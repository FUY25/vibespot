## Task 1: Fix Search Field Typing Bug

**Root cause:** `restoreSearchFieldFocus()` calls `panel.makeFirstResponder(searchField)` on every `tableViewSelectionDidChange`. When `applyResults` selects row 0, this fires, and `makeFirstResponder` on a text field selects all text — so the next keystroke replaces everything.

**Files:**
- Modify: `Sources/VibeLight/UI/SearchPanelController.swift`

- [ ] **Step 1: Write failing test**

```swift
@Test @MainActor
func testSearchFieldRetainsInsertionPointAfterResultsRefresh() throws {
    let controller = SearchPanelController()
    let dbPath = "/tmp/vibelight_focus_test_\(UUID().uuidString).sqlite3"
    defer { try? FileManager.default.removeItem(atPath: dbPath) }

    let index = try SessionIndex(dbPath: dbPath)
    try index.upsertSession(
        id: "s1", tool: "claude", title: "test session",
        project: "/p", projectName: "proj", gitBranch: "main",
        status: "live", startedAt: Date(), pid: 1
    )
    controller.sessionIndex = index
    controller.show()

    // Simulate typing "he" — the field should not auto-select
    let searchField = controller.exposedSearchField
    searchField.stringValue = "he"
    // Move cursor to end
    if let editor = searchField.currentEditor() {
        editor.selectedRange = NSRange(location: 2, length: 0)
    }

    // Trigger results refresh (which causes table selection change)
    controller.simulateControlTextDidChange()

    // After refresh, cursor should still be at position 2, not selecting all
```

- [ ] **Step 2: Fix restoreSearchFieldFocus to not select all text**

Replace `restoreSearchFieldFocus()` in `SearchPanelController.swift`:

```swift
private func restoreSearchFieldFocus() {
    guard panel.isVisible else { return }

    // Only restore focus if the search field lost it.
    // Do NOT call makeFirstResponder — that selects all text.
    if panel.firstResponder != searchField.currentEditor() &&
       panel.firstResponder != searchField {
        DispatchQueue.main.async { [weak self] in
            guard let self else { return }
            self.panel.makeFirstResponder(self.searchField)
            // Restore cursor to end instead of selecting all
            if let editor = self.searchField.currentEditor() {
                let length = self.searchField.stringValue.count
                editor.selectedRange = NSRange(location: length, length: 0)
            }
        }
    }
}
```

```swift
// Sources/VibeLight/UI/ToolIcon.swift
import AppKit

enum ToolIcon {
    static func image(for tool: String, size: CGFloat = 20) -> NSImage {
        let name: String
        switch tool.lowercased() {
        case "claude": name = "claude-icon"
        case "codex":  name = "codex-icon"
        case "gemini": name = "gemini-icon"
        default:       return fallbackImage(for: tool, size: size)
        }

        if let image = NSImage(named: name) {
            image.size = NSSize(width: size, height: size)
            return image
        }

        return fallbackImage(for: tool, size: size)
    }

    /// Fallback: colored circle with first letter (only used if asset is missing)
    private static func fallbackImage(for tool: String, size: CGFloat) -> NSImage {
        let image = NSImage(size: NSSize(width: size, height: size), flipped: false) { rect in
            NSColor.systemGray.setFill()
            NSBezierPath(ovalIn: rect.insetBy(dx: 1, dy: 1)).fill()

            let letter = String(tool.prefix(1)).uppercased()
            let attrs: [NSAttributedString.Key: Any] = [
                .font: NSFont.systemFont(ofSize: size * 0.48, weight: .bold),
                .foregroundColor: NSColor.white,
            ]
            let attrStr = NSAttributedString(string: letter, attributes: attrs)
            let strSize = attrStr.size()
            attrStr.draw(at: NSPoint(
                x: rect.midX - strSize.width / 2,
                y: rect.midY - strSize.height / 2
            ))
            return true
        }
        image.isTemplate = false
        return image
    }
}
```

```swift
import Foundation

enum RelativeTimeFormatter {
    static func string(from date: Date, relativeTo now: Date = Date()) -> String {
        let interval = now.timeIntervalSince(date)

        if interval < 60 {
            return "just now"
        } else if interval < 3600 {
            let minutes = Int(interval / 60)
            return "\(minutes)m ago"
        } else if interval < 86400 {
            let hours = Int(interval / 3600)
            return "\(hours)h ago"
        } else if interval < 172800 {
            return "yesterday"
        } else if interval < 604800 {
            let days = Int(interval / 86400)
            return "\(days)d ago"
        } else {
            let formatter = DateFormatter()
            formatter.dateStyle = .medium
            formatter.timeStyle = .none
            return formatter.string(from: date)
        }
    }
}
```

```swift
private func makeMetadataText(for result: SearchResult) -> String {
    let time = RelativeTimeFormatter.string(from: result.lastActivityAt)
    let projectName = result.projectName.isEmpty
        ? URL(fileURLWithPath: result.project).lastPathComponent
        : result.projectName
    let branch = result.gitBranch.trimmingCharacters(in: .whitespacesAndNewlines)
    let tokens = formatTokenCount(result.tokenCount)

    var parts = [time]
    if !projectName.isEmpty {
        let projectPart = branch.isEmpty ? projectName : "\(projectName) / \(branch)"
        parts.append(projectPart)
    }
    if result.tokenCount > 0 {

    return parts.joined(separator: " · ")
}

private func formatTokenCount(_ count: Int) -> String {
    if count >= 1000 {
```

Add a method to determine session activity status:

```swift
enum SessionActivityStatus: String {
    case working  // actively processing
    case waiting  // idle, awaiting user input
    case closed   // session ended
}

// In Indexer or a new helper:
static func activityStatus(
    sessionStatus: String,
    lastFileModification: Date,
    lastJSONLEntryType: String?,  // "tool_use", "assistant", "user", "tool_result"
    now: Date = Date()
) -> SessionActivityStatus {
    guard sessionStatus == "live" else {
        return .closed
    }

    let secondsSinceModification = now.timeIntervalSince(lastFileModification)

    // File actively being written to → Working
    if secondsSinceModification < 5 {
        return .working
    }

    // File quiet but last entry is a tool_use without result → Working (tool executing)
    if lastJSONLEntryType == "tool_use" {
        return .working
    }

    // File quiet, last entry is user message → Working (assistant processing)
    if lastJSONLEntryType == "user" {
        return .working
    }

    // File quiet, last entry is assistant or tool_result → Waiting
    return .waiting
}
```
