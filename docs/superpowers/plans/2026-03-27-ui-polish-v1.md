## Task 1: Fix Search Field Typing Bug

**Root cause:** `restoreSearchFieldFocus()` calls `panel.makeFirstResponder(searchField)` on every `tableViewSelectionDidChange`. When `applyResults` selects row 0, this fires, and `makeFirstResponder` on a text field selects all text — so the next keystroke replaces everything.

**Files:**
- Modify: `Sources/VibeLight/UI/SearchPanelController.swift`

- [ ] **Step 1: Write failing test**

```swift
// Tests/VibeLightTests/SearchFieldFocusTests.swift
import Testing
import AppKit
@testable import VibeLight

@Test @MainActor
func testSearchFieldRetainsInsertionPointAfterResultsRefresh() throws {
    let controller = SearchPanelController()
    let dbPath = "/tmp/vibelight_focus_test_\(UUID().uuidString).sqlite3"
    defer { try? FileManager.default.removeItem(atPath: dbPath) }

    let index = try SessionIndex(dbPath: dbPath)
    try index.upsertSession(
        id: "s1", tool: "claude", title: "test session",
        project: "/p", projectName: "proj", gitBranch: "main",
        status: "live", startedAt: Date(), pid: 1
    )
    controller.sessionIndex = index
    controller.show()

    // Simulate typing "he" — the field should not auto-select
    let searchField = controller.exposedSearchField
    searchField.stringValue = "he"
    // Move cursor to end
    if let editor = searchField.currentEditor() {
        editor.selectedRange = NSRange(location: 2, length: 0)
    }

    // Trigger results refresh (which causes table selection change)
    controller.simulateControlTextDidChange()

    // After refresh, cursor should still be at position 2, not selecting all
    if let editor = searchField.currentEditor() {
        #expect(editor.selectedRange.length == 0, "Text should NOT be selected after results refresh")
    }

    controller.hide()
}
```

- [ ] **Step 2: Fix restoreSearchFieldFocus to not select all text**

Replace `restoreSearchFieldFocus()` in `SearchPanelController.swift`:

```swift
private func restoreSearchFieldFocus() {
    guard panel.isVisible else { return }

    // Only restore focus if the search field lost it.
    // Do NOT call makeFirstResponder — that selects all text.
    if panel.firstResponder != searchField.currentEditor() &&
       panel.firstResponder != searchField {
        DispatchQueue.main.async { [weak self] in
            guard let self else { return }
            self.panel.makeFirstResponder(self.searchField)
            // Restore cursor to end instead of selecting all
            if let editor = self.searchField.currentEditor() {
                let length = self.searchField.stringValue.count
                editor.selectedRange = NSRange(location: length, length: 0)
            }
        }
    }
}
```

- [ ] **Step 1: Create ToolIcon that renders colored circles with SF Symbol overlays**

Use programmatic drawing — no bundled assets needed. Each tool gets a distinct color and SF Symbol.

```swift
// Sources/VibeLight/UI/ToolIcon.swift
import AppKit

enum ToolIcon {
    static func image(for tool: String, size: CGFloat = 20) -> NSImage {
        let image = NSImage(size: NSSize(width: size, height: size), flipped: false) { rect in
            let color = iconColor(for: tool)
            color.setFill()
            NSBezierPath(ovalIn: rect.insetBy(dx: 1, dy: 1)).fill()

            // Draw first letter centered
            let letter = String(tool.prefix(1)).uppercased()
            let attrs: [NSAttributedString.Key: Any] = [
                .font: NSFont.systemFont(ofSize: size * 0.48, weight: .bold),
                .foregroundColor: NSColor.white,
            ]
            let attrStr = NSAttributedString(string: letter, attributes: attrs)
            let strSize = attrStr.size()
            let origin = NSPoint(
                x: rect.midX - strSize.width / 2,
                y: rect.midY - strSize.height / 2
            )
            attrStr.draw(at: origin)
            return true
        }
        image.isTemplate = false
        return image
    }

    private static func iconColor(for tool: String) -> NSColor {
        switch tool.lowercased() {
        case "claude": return NSColor(red: 0.82, green: 0.45, blue: 0.24, alpha: 1.0)  // Claude orange
        case "codex":  return NSColor(red: 0.20, green: 0.75, blue: 0.40, alpha: 1.0)  // OpenAI green
        case "gemini": return NSColor(red: 0.30, green: 0.52, blue: 0.90, alpha: 1.0)  // Google blue
        default:       return NSColor.systemGray
        }
    }
}
```

## Task 5: Spotlight-Faithful UI Redesign

Reference: macOS Spotlight (Cmd+Space) uses these exact specs:
- **Panel width:** ~680px
- **Corner radius:** 12px (not 20)
- **Search field:** Plain text field (not NSSearchField), ~28px font, SF Pro, no border, no focus ring, no magnifying glass built into field — the icon is separate
- **Vibrancy:** `.sidebar` material (lighter than `.hudWindow`)
- **Results:** No intercell spacing. Rows are flush. Selected row uses accent-colored rounded rect highlight.
- **Row height:** ~36-40px for simple rows
- **Positioning:** Upper third of screen (not centered), roughly 200px from top of screen
- **No visible border** on the panel
- **Thin separator line** between search field and results
- **Max ~9 visible results** before scroll

- [ ] **Step 1: Update SearchField to plain NSTextField style**

Replace `NSSearchField` with a plain `NSTextField` that looks like Spotlight's input:

```swift
// SearchField.swift
import AppKit

final class SearchField: NSTextField {
    override init(frame frameRect: NSRect) {
        super.init(frame: frameRect)
        configure()
    }

    required init?(coder: NSCoder) {
        super.init(coder: coder)
        configure()
    }

    override var intrinsicContentSize: NSSize {
        var size = super.intrinsicContentSize
        size.height = max(size.height, 36)
        return size
    }

    private func configure() {
        translatesAutoresizingMaskIntoConstraints = false
        placeholderString = "Search sessions"
        focusRingType = .none
        isBordered = false
        drawsBackground = false
        font = .systemFont(ofSize: 24, weight: .light)
        textColor = .labelColor
        usesSingleLineMode = true
        lineBreakMode = .byTruncatingTail
        cell?.sendsActionOnEndEditing = false
    }
}
```

Note: Changing from `NSSearchField` to `NSTextField` means `SearchPanelController` should change its delegate conformance from `NSSearchFieldDelegate` to `NSTextFieldDelegate`. Both support `controlTextDidChange` and `control(_:textView:doCommandBy:)`.
